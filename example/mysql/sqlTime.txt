include "@/datetime/date/parseDate"
include "@/datetime/isoString"
include "@/datetime/time/parseTime"

func main
    conn = -1

    try
        conn = connect("tcp://localhost:3306", "root", "root")
    catch err
        echo "SQL: {}" format err
        
        return
    end try

    try
        res = query(conn, "select current_timestamp")
        res = split((res cell 1, 0), " ")
        
        timestamp = parseDate(first res) + parseTime(last res)
        
        echo "{}\n" format isoString(timestamp)
    catch err
        echo "SQL: {}" format err
    end try

    try
        close(conn)
    catch err
        echo "SQL: {}" format err
    end try
end func

main()
