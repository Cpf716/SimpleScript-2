func main
    conn = -1

    try
        conn = connect("tcp://localhost:3306", "root", "root")
    catch err
        echo "SQL: {}\n" format err
        
        return
    end try

    try
        array res = prepareQuery(conn, "select * from music_library.artists")
        
        for i = 0, i < (count res - 1) / first res, i += 1
            for j = 0, j < first res, j += 1
                echo "{}\t" format (res cell i, j)
            end for
            
            echo "\n"
        end for
    catch err
        echo "SQL: {}\n" format err
    end try

    try
        close(conn)
    catch err
        echo "SQL: {}\n" format err
    end try
end func

main()
