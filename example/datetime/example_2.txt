include ((dir = "@/datetime/") + "dateString")
include (dir + "parseTime")
include (dir + "transform/addWeeks")

func main
    if nrows(argv) !== 1
        expectError("0 argument(s), got {}" format (nrows(argv) - 1))
    end if

    dateStr = dateString(addWeeks(parseTime(local()), 1))

    echo "{}\n" format dateStr
end func

main()
