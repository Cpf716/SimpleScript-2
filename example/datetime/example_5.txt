include "@/datetime/time/meridianTime"
include "@/search/string/split"
include "@/datetime/parseTime"

func main
    if nrows(argv) !== 1
        expectError("0 argument(s), got {}" format (nrows(argv) - 1))
    end if
        
    time = meridianTime(parseTime(local()))

    arr = split(time, " ")

    arr.0 = ((first arr) slice 0, 5)

    time = arr join " "

    echo "{}\n" format time
end func

main()
