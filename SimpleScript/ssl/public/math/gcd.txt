include "@/error/expectError"

func main
    if nrows(argv) !== 3
        expectError("2 argument(s), got {}" format (nrows(argv) - 1))
    end if

    a = ((argv row 1) slice 1, argv cell 0, 0)

    if subtypeOf a !== "int"
        typeError(subtypeOf a, "int")
    end if

    b = ((argv row 2) slice 1, argv cell 1, 0)

    if subtypeOf b !== "int"
        typeError(subtypeOf b, "int")
    end if

    if a < b
        tmp = a
        a = b
        b = tmp
    end if

    r = 0
    do while r !== 0
        r = a % b
        a = b
        b = r
    end while

    return a
end func

return main()
