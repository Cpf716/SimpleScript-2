include "@/error/expectError"

if nrows(argv) !== 2 && nrows(argv) !== 3
    expectError("1 argument(s), got {}" format (nrows(argv) - 1))
end if

a = (argv row 1) slice 1

if subtypeOf a !== "int"
    typeError(subtypeOf a, "int")
end if

if nrows(argv) === 2
    factorial(a, a - 1)
else
    b = (argv row 2) slice 1

    if b === 1
        return a
    end if
    
    factorial(a * b, b - 1)
end if
