include "@/sort/sort"

func main
    if (count argv - 1) / first argv !== 2
        expectError("1 argument(s), got {}" format ((count argv - 1) / first argv - 1))
    end if

    array arr = (argv row 1) slice 1

    arr = sort(arr)

    return arr.floor(count arr / 2)
end func

return main()
