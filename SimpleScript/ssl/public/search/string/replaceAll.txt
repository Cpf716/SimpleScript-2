include "@/error/expectError"

func main
    if nrows(argv) !== 4
        expectError("3 argument(s), got {}" format (nrows(argv) - 1))
    end if

    text = ((argv row 1) slice 1, argv cell 1, 0)

    if typeOf text !== "string"
        typeError(typeOf text, "string")
    end if

    if text === null
        throw "null"
    end if

    pattern = ((argv row 2) slice 1, argv cell 2, 0)

    if typeOf pattern !== "string"
        typeError(typeOf pattern, "string")
    end if

    if pattern === null
        throw "null"
    end if

    newPattern = ((argv row 3) slice 1, argv cell 3, 0)

    if typeOf newPattern !== "string"
        typeError(typeOf newPattern, "string")
    end if

    if newPattern === null
        throw "null"
    end if

    for i = 0, i <= count text - count pattern,
        j = 0
        while j < count pattern && text.(i + j) === pattern.j
            j += 1
        end while
        
        if j === count pattern
            text = (text slice 0, i) + newPattern + (text substr i + count pattern)
            i += (count newPattern ? count newPattern, 1)
        end if
    end for

    return text
end func

return main()
