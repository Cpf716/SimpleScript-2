include "@/error/expectError"

func main
    if nrows(argv) !== 3
        expectError("2 argument(s), got {}" format (nrows(argv) - 1))
    end if

    text = ((argv row 1) slice 1, argv cell 1, 0)

    if typeOf text !== "string"
        typeError(typeOf text, "string")
    end if

    if text === null
        throw "null"
    end if

    pattern = ((argv row 2) slice 1, argv cell 2, 0)

    if typeOf pattern !== "string"
        typeError(typeOf pattern, "string")
    end if

    if pattern === null
        throw "null"
    end if

    i = count text - count pattern
    for , i >= 0, i -= 1
        j = 0
        while j < count pattern && text.(i + j) === pattern.j
            j += 1
        end while
        
        if j === count pattern
            break
        end if
    end for

    return i
end fun

return main()
