include "@/exception/expectExcept"

func main
    if nrows(argv) !== 2
        expectExcept("1 argument(s), got {}" format (nrows(argv) - 1))
    end if

    text = (argv row 1) slice 1

    if typeOf text !== "string"
        typeExcept(typeOf text, "string")
    end if

    if text === null
        throw "null"
    end if

    i = 0
    while i < count text && isLower text.i
        i += 1
    end while

    return i === count text
end func

return main()
