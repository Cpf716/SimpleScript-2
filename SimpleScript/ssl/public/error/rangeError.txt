include "@/error/expectError"

if nrows(argv) !== 2
    expectError("1 argument(s), got {}" format (nrows(argv) - 1))
end if

message = (argv row 1) slice 1

if typeOf message !== "string"
    typeError(typeOf message, "string")
end if

throw "Out of range: {}" format message
