include "@/error/typeError"

if nrows(argv) !== 2
    throw "Expected 2 argument(s), got {}" format (nrows(argv) - 1)
end if

message = (argv row 1) slice 1

if typeOf message !== "string"
    typeError(typeOf message, "string")
end if

throw "Expected {}" format message
