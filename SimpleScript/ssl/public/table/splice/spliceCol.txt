include "@/error/expectError"

func main
    if (count argv - 1) / first argv !== 3
        expectError("2 argument(s), got {}" format ((count argv - 1) / first argv - 1))
    end if

    arr = ((argv row 1) slice 1, argv cell 1, 0)

    if subtypeOf arr !== "table"
        typeError(subtypeOf arr, "table")
    end if

    idx = ((argv row 2) slice 1, argv cell 2, 0)

    if subtypeOf idx !== "int"
        typeError(subtypeOf idx, "int")
    end if

    if idx < 0 || idx >= first arr
        rangeError("idx {}, ncols {}" format array(idx, first arr))
    end if

    arr = inverse arr

    arr splice idx + 1, first arr

    return inverse arr
end func

return main()
