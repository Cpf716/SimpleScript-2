include "@/error/expectError"

func main
    if nrows(argv) !== 2
        expectError("1 argument(s), got {}" format (nrows(argv) - 1))
    end if

    year = (argv row 1) slice 1

    if subtypeOf year !== "int"
        typeError(subtypeOf year, "int")
    end if

    if year < 1970
        return
    end if

    return year % 4 ? false, year % 100 ? true, !(year % 400)
end func

return main()
