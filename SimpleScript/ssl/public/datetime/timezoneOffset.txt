include "@/datetime/parseTime"

func main
    if nrows(argv) !== 1
        expectError("0 argument(s), got {}" format (nrows(argv) - 1))
    end if

    end = local(), beg = gmt()

    `timezoneOffset` = parseTime(end) - parseTime(beg)
    `timezoneOffset` = (`timezoneOffset` - `timezoneOffset` % 60) / 60
    
    return `timezoneOffset`
end func

return main()
