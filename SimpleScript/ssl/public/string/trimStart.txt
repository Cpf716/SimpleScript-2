include "@/error/expectError"

func main
    if nrows(argv) !== 2
        expectError("1 argument(s), got {}" format (nrows(argv) - 1))
    end if

    text = (argv row 1) slice 1

    if typeOf text !== "string"
        typeError(typeOf text, "string")
    end if

    if text === null
        throw "null"
    end if

    s = 0
    while s < count text && isSpace text.s
        s += 1
    end while

    return text substr s
end func

return main()
