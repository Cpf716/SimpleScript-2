include "@/error/expectError"

func main
    if nrows(argv) !== 2
        expectError("1 argument(s), got {}" format (nrows(argv) - 1))
    end if

    text = (argv row 1) slice 1

    if typeOf text !== "string"
        typeError(typeOf text, "string")
    end if

    if text === null
        throw "null"
    end if

    e = count text
    while e > 0 && isSpace text.(e - 1)
        e -= 1
    end while

    return text slice 0, e
end func

return main()
