include "@/error/expectError"

func main
    if nrows(argv) !== 2
        expectError("1 argument(s), got {}" format (nrows(argv) - 1))
    end if

    val = (argv row 1) slice 1

    if typeOf val === "array"
        return null
    end if

    return "" + val
end func

return main()
