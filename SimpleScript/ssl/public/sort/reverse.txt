include "@/error/expectError"

func main
    if (count argv - 1) / first argv !== 2
        expectError("1 argument(s), got {}" format ((count argv - 1) / first argv - 1))
    end if

    array arr = (argv row 1) slice 1

    for i = 0, i < floor (count arr / 2), i += 1
        tmp = arr.i
        
        arr.i = arr.(count arr - i - 1)
        arr.(count arr - i - 1) = tmp
    end for

    return arr
end func

return main()
