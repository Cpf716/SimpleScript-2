include "@/error/expectError"

func main
    if (count argv - 1) / first argv !== 2
        expectError("1 argument(s), got {}" format ((count argv - 1) / first argv - 1))
    end if

    array arr = (argv row 1) slice 1

    for i = 0, i < count arr - 1, i += 1
        j = i + 1
        while j < count arr
            if arr.i === arr.j
                arr splice j
            else
                j += 1
            end if
        end while
    end for

    return arr
end func

return main()
